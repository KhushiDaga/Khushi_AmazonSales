import pandas as pd

# Load the dataset
# Replace 'sales_data.csv' with the actual file path of your dataset
sales_data = pd.read_csv('/Users/abc/Desktop/drive-download-20240210T115751Z-001/AmazonSalesData.csv')

# Data Transformation
# Convert date columns to datetime objects
sales_data['Order Date'] = pd.to_datetime(sales_data['Order Date'])
sales_data['Ship Date'] = pd.to_datetime(sales_data['Ship Date'])

# Calculate Total Revenue, Total Cost, and Total Profit
sales_data['Total Revenue'] = sales_data['Units Sold'] * sales_data['Unit Price']
sales_data['Total Cost'] = sales_data['Units Sold'] * sales_data['Unit Cost']
sales_data['Total Profit'] = sales_data['Total Revenue'] - sales_data['Total Cost']

# Extract year and month from Order Date
sales_data['Year'] = sales_data['Order Date'].dt.year
sales_data['Month'] = sales_data['Order Date'].dt.month

# Aggregating data for analysis
monthly_sales = sales_data.groupby(['Year', 'Month']).agg({
    'Total Revenue': 'sum',
    'Total Cost': 'sum',
    'Total Profit': 'sum'
}).reset_index()

yearly_sales = sales_data.groupby('Year').agg({
    'Total Revenue': 'sum',
    'Total Cost': 'sum',
    'Total Profit': 'sum'
}).reset_index()

yearly_monthly_sales = sales_data.groupby(['Year', 'Month']).agg({
    'Total Revenue': 'sum',
    'Total Cost': 'sum',
    'Total Profit': 'sum'
}).reset_index()

# Analysis and Insights
# Visualizations, statistical analysis, machine learning, etc. can be performed here

# Example of visualizing monthly sales trend
import matplotlib.pyplot as plt

plt.figure(figsize=(10, 6))
plt.plot(monthly_sales['Year'].astype(str) + '-' + monthly_sales['Month'].astype(str), monthly_sales['Total Revenue'], marker='o')

plt.ylabel('Total Revenue')
